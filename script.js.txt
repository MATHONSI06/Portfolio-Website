// Example: dynamically load projects
const projects = [
  { title: "Supermarket Management System", description: "Database project using SQL." },
  { title: "Java Swing Login Panel", description: "GUI project built in NetBeans." }
];

const projectList = document.getElementById("project-list");

projects.forEach(project => {
  const div = document.createElement("div");
  div.innerHTML = `<h3>${project.title}</h3><p>${project.description}</p>`;
  projectList.appendChild(div);
});
